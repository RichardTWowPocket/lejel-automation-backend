// ─── HELPERS ───
function getActiveWordIndex(t: number, words: typeof WHISPER_WORDS): number {
  for (let i = 0; i < words.length; i++) {
    if (words[i].start <= t && words[i].end > t) return i;
    if (words[i].start > t) return Math.max(0, i - 1);
  }
  return words.length - 1;
}

function buildChunks(
  words: typeof WHISPER_WORDS,
  maxPerLine = 4,
): Array<{ words: typeof WHISPER_WORDS; startIdx: number }> {
  const out: Array<{ words: typeof WHISPER_WORDS; startIdx: number }> = [];
  let i = 0;
  while (i < words.length) {
    let j = i + maxPerLine;
    if (j > words.length) j = words.length;
    for (let k = i; k < j; k++) {
      if (/[.?!]$/.test(words[k].word) && k >= i + 2) {
        j = k + 1;
        break;
      }
      if (k > i && words[k].start - words[k - 1].end > 0.55) {
        j = k;
        break;
      }
    }
    out.push({ words: words.slice(i, j), startIdx: i });
    i = j;
  }
  return out;
}

// ─── SEGMENT MOTION WRAPPER ───
function SegmentMotion({
  children,
  startFrame,
  voiceEndFrame,
  durationInFrames,
}: {
  children: React.ReactNode;
  startFrame: number;
  voiceEndFrame: number;
  durationInFrames: number;
}) {
  const frame = useCurrentFrame();
  const { fps } = useVideoConfig();
  const local = Math.max(0, frame - startFrame);
  const enter = spring({ frame: local, fps, config: { damping: 20, stiffness: 96 } });
  const voiceEndLocal = Math.max(0, voiceEndFrame - startFrame);
  const exitStart = Math.min(
    Math.max(0, voiceEndLocal),
    Math.max(0, durationInFrames - 2),
  );
  const exitLen = Math.min(24, Math.max(10, durationInFrames - exitStart));
  const exitProg = interpolate(
    local,
    [exitStart, exitStart + exitLen],
    [0, 1],
    { extrapolateLeft: "clamp", extrapolateRight: "clamp" },
  );
  const groupOpacity = enter * (1 - exitProg * 0.98);
  const groupScale = 0.93 + enter * 0.09 - exitProg * 0.1;
  const groupY = (1 - enter) * 22 + exitProg * 18;

  return (
    <div
      style={{
        opacity: groupOpacity,
        transform: `translateY(${groupY}px) scale(${groupScale})`,
        willChange: "transform, opacity",
        width: "100%",
        height: "100%",
      }}
    >
      {children}
    </div>
  );
}

// ─── CAPTIONS ───
function WhisperSocialCaptions() {
  const frame = useCurrentFrame();
  const { fps } = useVideoConfig();
  const t = frame / fps;
  const activeIdx = getActiveWordIndex(t, WHISPER_WORDS);
  const chunks = buildChunks(WHISPER_WORDS, 4);

  let currentChunkIndex = 0;
  let cumWords = 0;
  for (let c = 0; c < chunks.length; c++) {
    const nextCum = cumWords + chunks[c].words.length;
    if (activeIdx >= cumWords && activeIdx < nextCum) {
      currentChunkIndex = c;
      break;
    }
    cumWords = nextCum;
  }
  const chunk = chunks[currentChunkIndex] ?? { words: [], startIdx: 0 };
  const chunkStart = chunk.startIdx;

  return (
    <div
      style={{
        position: "absolute",
        bottom: 0,
        left: 0,
        right: 0,
        display: "flex",
        justifyContent: "center",
        alignItems: "flex-end",
        paddingBottom: 200,
        pointerEvents: "none",
      }}
    >
      <div
        style={{
          background: "rgba(0,0,0,0.62)",
          borderRadius: 20,
          padding: "18px 22px",
          maxWidth: "90%",
          display: "flex",
          flexWrap: "wrap",
          gap: 6,
          justifyContent: "center",
        }}
      >
        {chunk.words.map((w, i) => {
          const globalIdx = chunkStart + i;
          const isActive = globalIdx === activeIdx;
          const isPast = globalIdx < activeIdx;
          return (
            <span
              key={globalIdx}
              style={{
                fontFamily: FONT_STACK,
                fontSize: 32,
                fontWeight: isActive ? 900 : 600,
                color: isActive
                  ? "#fef08a"
                  : isPast
                    ? "rgba(148,163,184,0.55)"
                    : "rgba(248,250,252,0.5)",
                transition: "none",
                transform: isActive ? "scale(1.08)" : "scale(1)",
              }}
            >
              {w.word}
            </span>
          );
        })}
      </div>
    </div>
  );
}

// ─── MAIN COMPOSITION ───
export default function Composition() {
  const { audioUrl } = getInputProps() as { audioUrl?: string };
  return (

    >
      {audioUrl ? <Audio src={audioUrl} /> : null}
      SCENE_SEQUENCE_PLACEHOLDER
      <Sequence from={0} durationInFrames={TOTAL_FRAMES}>
        <WhisperSocialCaptions />
      </Sequence>
    </AbsoluteFill>
  );
}
